103 /**
104  * The definition of a helios type struct. This is the part in any helios
105  * object defining what type it is. Any helios struct must define a type and use
106  * it in the HELIOS_OBJECT_COMMON_BASE which tells the object system what type
107  * the object is. It has pointers to all the functions and fields acting on the
108  * type which are standardized. Non standard functions can still be exported
109  * (example: helios_string_from_charp).
110  *
111  * Inheritance system:
112  * Any object has an helios_type object in it of which the object is an
113  * instance. helios_type is always instance of itself. By putting a pointer to
114  * the type of an object in the first field of the struct, any helios_object can
115  * be cast to the msot basic object possible (helios_object), and using the type
116  * it's type can be determined and the object can be cast back. This makes it
117  * possible to use helios_object as type for any helios object.
118  */
119 typedef struct __helios_type_s {
120  HELIOS_OBJECT_COMMON_BASE;
121  HELIOS_OBJECT_BASIC_FIELDS;
122  HELIOS_OBJECT_BINOPS_FIELDS;
123 } helios_type;
